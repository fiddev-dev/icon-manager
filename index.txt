    <script>
        let canvas = document.getElementById('iconCanvas');
        let ctx = canvas.getContext('2d');
        let uploadedImage = null;
        let downloadSize = 512;
        let isGradient = false;
        let availableIcons = [];
        let allIconsData = [];

        canvas.width = 400;
        canvas.height = 400;

        function handleImageUpload(event) {
            const file = event.target.files[0];
            if (file && file.type === 'image/png') {
                const reader = new FileReader();
                reader.onload = function(e) {
                    const img = new Image();
                    img.onload = function() {
                        uploadedImage = img;
                        document.getElementById('uploadPrompt').style.display = 'none';
                        updateCanvas();
                    };
                    img.src = e.target.result;
                };
                reader.readAsDataURL(file);
            } else {
                alert('Silakan upload file PNG!');
            }
        }

        function toggleGradient() {
            isGradient = document.getElementById('gradientToggle').checked;
            if (isGradient) {
                document.getElementById('solidColorGroup').style.display = 'none';
                document.getElementById('gradientColorGroup').style.display = 'block';
            } else {
                document.getElementById('solidColorGroup').style.display = 'block';
                document.getElementById('gradientColorGroup').style.display = 'none';
            }
            updateCanvas();
        }

        function updateCanvas() {
            if (!uploadedImage) return;

            const bgColor = document.getElementById('bgColor').value;
            const gradientColor1 = document.getElementById('gradientColor1').value;
            const gradientColor2 = document.getElementById('gradientColor2').value;
            const borderRadius = document.getElementById('borderRadius').value;
            const hueRotate = document.getElementById('hueRotate').value;
            const brightness = document.getElementById('brightness').value;
            const padding = document.getElementById('padding').value;
            const shadowIntensity = document.getElementById('shadowIntensity').value;
            const inverse = document.getElementById('inverseToggle').checked;

            document.getElementById('radiusValue').textContent = borderRadius + '%';
            document.getElementById('hueValue').textContent = hueRotate + '¬∞';
            document.getElementById('brightnessValue').textContent = brightness + '%';
            document.getElementById('paddingValue').textContent = padding + '%';
            document.getElementById('shadowValue').textContent = shadowIntensity + 'px';

            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.save();

            const radius = (borderRadius / 100) * Math.min(canvas.width, canvas.height) / 2;
            
            if (borderRadius >= 100) {
                ctx.beginPath();
                ctx.arc(canvas.width / 2, canvas.height / 2, canvas.width / 2, 0, Math.PI * 2);
                ctx.closePath();
            } else {
                ctx.beginPath();
                ctx.moveTo(radius, 0);
                ctx.lineTo(canvas.width - radius, 0);
                ctx.arcTo(canvas.width, 0, canvas.width, radius, radius);
                ctx.lineTo(canvas.width, canvas.height - radius);
                ctx.arcTo(canvas.width, canvas.height, canvas.width - radius, canvas.height, radius);
                ctx.lineTo(radius, canvas.height);
                ctx.arcTo(0, canvas.height, 0, canvas.height - radius, radius);
                ctx.lineTo(0, radius);
                ctx.arcTo(0, 0, radius, 0, radius);
                ctx.closePath();
            }

            // Shadow
            if (shadowIntensity > 0) {
                ctx.shadowColor = 'rgba(0, 0, 0, 0.3)';
                ctx.shadowBlur = shadowIntensity;
                ctx.shadowOffsetX = shadowIntensity / 4;
                ctx.shadowOffsetY = shadowIntensity / 4;
            }

            // Background
            if (isGradient) {
                const gradient = ctx.createLinearGradient(0, 0, canvas.width, canvas.height);
                gradient.addColorStop(0, gradientColor1);
                gradient.addColorStop(1, gradientColor2);
                ctx.fillStyle = gradient;
            } else {
                ctx.fillStyle = bgColor;
            }
            ctx.fill();

            // Reset shadow for image
            ctx.shadowColor = 'transparent';
            ctx.shadowBlur = 0;
            ctx.shadowOffsetX = 0;
            ctx.shadowOffsetY = 0;

            ctx.clip();

            const invertFilter = inverse ? 'invert(1) ' : '';
            ctx.filter = `${invertFilter}hue-rotate(${hueRotate}deg) brightness(${brightness}%)`;

            const paddingPercent = padding / 100;
            const scale = Math.min(canvas.width / uploadedImage.width, canvas.height / uploadedImage.height) * (1 - paddingPercent);
            const x = (canvas.width - uploadedImage.width * scale) / 2;
            const y = (canvas.height - uploadedImage.height * scale) / 2;
            
            ctx.drawImage(uploadedImage, x, y, uploadedImage.width * scale, uploadedImage.height * scale);

            ctx.restore();
        }

        function setDownloadSize(size) {
            downloadSize = size;
            document.querySelectorAll('.size-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
        }

        function renderIconToCanvas(tempCanvas, size) {
            const tempCtx = tempCanvas.getContext('2d');
            tempCanvas.width = size;
            tempCanvas.height = size;

            const bgColor = document.getElementById('bgColor').value;
            const gradientColor1 = document.getElementById('gradientColor1').value;
            const gradientColor2 = document.getElementById('gradientColor2').value;
            const borderRadius = document.getElementById('borderRadius').value;
            const hueRotate = document.getElementById('hueRotate').value;
            const brightness = document.getElementById('brightness').value;
            const padding = document.getElementById('padding').value;
            const shadowIntensity = document.getElementById('shadowIntensity').value;

            tempCtx.save();

            const radius = (borderRadius / 100) * Math.min(size, size) / 2;
            
            if (borderRadius >= 100) {
                tempCtx.beginPath();
                tempCtx.arc(size / 2, size / 2, size / 2, 0, Math.PI * 2);
                tempCtx.closePath();
            } else {
                tempCtx.beginPath();
                tempCtx.moveTo(radius, 0);
                tempCtx.lineTo(size - radius, 0);
                tempCtx.arcTo(size, 0, size, radius, radius);
                tempCtx.lineTo(size, size - radius);
                tempCtx.arcTo(size, size, size - radius, size, radius);
                tempCtx.lineTo(radius, size);
                tempCtx.arcTo(0, size, 0, size - radius, radius);
                tempCtx.lineTo(0, radius);
                tempCtx.arcTo(0, 0, radius, 0, radius);
                tempCtx.closePath();
            }

            if (shadowIntensity > 0) {
                const shadowScale = size / 400;
                tempCtx.shadowColor = 'rgba(0, 0, 0, 0.3)';
                tempCtx.shadowBlur = shadowIntensity * shadowScale;
                tempCtx.shadowOffsetX = (shadowIntensity / 4) * shadowScale;
                tempCtx.shadowOffsetY = (shadowIntensity / 4) * shadowScale;
            }

            if (isGradient) {
                const gradient = tempCtx.createLinearGradient(0, 0, size, size);
                gradient.addColorStop(0, gradientColor1);
                gradient.addColorStop(1, gradientColor2);
                tempCtx.fillStyle = gradient;
            } else {
                tempCtx.fillStyle = bgColor;
            }
            tempCtx.fill();

            tempCtx.shadowColor = 'transparent';
            tempCtx.shadowBlur = 0;
            tempCtx.shadowOffsetX = 0;
            tempCtx.shadowOffsetY = 0;

            tempCtx.clip();

            const invertFilter = inverse ? 'invert(1) ' : '';
            tempCtx.filter = `${invertFilter}hue-rotate(${hueRotate}deg) brightness(${brightness}%)`;

            const paddingPercent = padding / 100;
            const scale = Math.min(size / uploadedImage.width, size / uploadedImage.height) * (1 - paddingPercent);
            const x = (size - uploadedImage.width * scale) / 2;
            const y = (size - uploadedImage.height * scale) / 2;
            
            tempCtx.drawImage(uploadedImage, x, y, uploadedImage.width * scale, uploadedImage.height * scale);
            tempCtx.restore();
        }

        function downloadIcon() {
            if (!uploadedImage) {
                alert('Silakan upload gambar terlebih dahulu!');
                return;
            }

            const tempCanvas = document.createElement('canvas');
            renderIconToCanvas(tempCanvas, downloadSize);

            tempCanvas.toBlob(function(blob) {
                const url = URL.createObjectURL(blob);
                const link = document.createElement('a');
                link.download = `icon-${downloadSize}x${downloadSize}.png`;
                link.href = url;
                link.click();
                URL.revokeObjectURL(url);
            });
        }

        async function executeBatchExport() {
            const checkboxes = document.querySelectorAll('.size-checklist input[type="checkbox"]:checked');
            
            if (checkboxes.length === 0) {
                alert('Pilih minimal satu ukuran untuk diexport!');
                return;
            }

            const tempCanvas = document.createElement('canvas');
            
            for (let checkbox of checkboxes) {
                const size = parseInt(checkbox.value);
                renderIconToCanvas(tempCanvas, size);
                
                const blob = await new Promise(resolve => tempCanvas.toBlob(resolve));
                const url = URL.createObjectURL(blob);
                const link = document.createElement('a');
                link.download = `icon-${size}x${size}.png`;
                link.href = url;
                link.click();
                URL.revokeObjectURL(url);
                
                await new Promise(resolve => setTimeout(resolve, 300));
            }

            alert(`Berhasil export ${checkboxes.length} icon!`);
            closeBatchExport();
        }

        async function loadBootstrapIcons() {
            const iconGrid = document.getElementById('iconGrid');
            iconGrid.innerHTML = '<div class="icon-loading">Memuat icon dari folder...</div>';

            try {
                // Coba baca file dari folder bootstrap-icons-1.13.1
                const commonIcons = [
                    'alarm', 'arrow-up', 'arrow-down', 'arrow-left', 'arrow-right',
                    'bell', 'bookmark', 'calendar', 'camera', 'cart',
                    'check', 'clock', 'cloud', 'download', 'edit',
                    'envelope', 'eye', 'file', 'folder', 'gear',
                    'heart', 'home', 'image', 'info-circle', 'lightbulb',
                    'link', 'list', 'lock', 'map', 'music',
                    'pencil', 'person', 'phone', 'play', 'plus',
                    'printer', 'search', 'share', 'star', 'trash',
                    'upload', 'user', 'video', 'wifi', 'x'
                ];

                // Coba load icon dari folder
                const iconPromises = commonIcons.map(async (iconName) => {
                    try {
                        const response = await fetch(`bootstrap-icons-1.13.1/icons/${iconName}.svg`);
                        if (response.ok) {
                            const svgText = await response.text();
                            return { name: iconName, svg: svgText, path: `bootstrap-icons-1.13.1/icons/${iconName}.svg` };
                        }
                    } catch (e) {
                        return null;
                    }
                    return null;
                });

                const loadedIcons = await Promise.all(iconPromises);
                availableIcons = loadedIcons.filter(icon => icon !== null);
                allIconsData = [...availableIcons];

                if (availableIcons.length > 0) {
                    displayIcons(availableIcons);
                } else {
                    // Fallback: tampilkan pesan untuk menggunakan CDN Bootstrap Icons
                    iconGrid.innerHTML = `
                        <div style="grid-column: 1/-1; text-align: center; padding: 40px;">
                            <p style="color: #999; margin-bottom: 20px;">
                                Folder "bootstrap-icons-1.13.1" tidak ditemukan atau kosong.<br>
                                Silakan pastikan folder ada di lokasi yang sama dengan file HTML ini.
                            </p>
                            <p style="color: #666;">
                                Atau gunakan fitur Upload Gambar PNG untuk mengedit icon Anda.
                            </p>
                        </div>
                    `;
                }
            } catch (error) {
                iconGrid.innerHTML = `
                    <div style="grid-column: 1/-1; text-align: center; padding: 40px;">
                        <p style="color: #ff6b6b; margin-bottom: 10px;">‚ùå Gagal memuat icon library</p>
                        <p style="color: #999; font-size: 0.9em;">
                            Pastikan folder "bootstrap-icons-1.13.1/icons/" tersedia di lokasi yang sama dengan file HTML
                        </p>
                    </div>
                `;
            }
        }

        function displayIcons(icons) {
            const iconGrid = document.getElementById('iconGrid');
            const iconCount = document.getElementById('iconCount');
            
            if (icons.length === 0) {
                iconGrid.innerHTML = '<div style="grid-column: 1/-1; text-align: center; padding: 20px; color: #999;">Tidak ada icon yang ditemukan</div>';
                iconCount.textContent = '0 icon ditemukan';
                return;
            }

            iconCount.textContent = `${icons.length} icon tersedia`;
            
            iconGrid.innerHTML = icons.map(icon => `
                <div class="icon-item" onclick="selectIcon('${icon.path}')" title="${icon.name}">
                    ${icon.svg}
                </div>
            `).join('');
        }

        async function selectIcon(iconPath) {
            try {
                const response = await fetch(iconPath);
                const svgText = await response.text();
                
                // Convert SVG to PNG using canvas
                const img = new Image();
                const svgBlob = new Blob([svgText], { type: 'image/svg+xml' });
                const url = URL.createObjectURL(svgBlob);
                
                img.onload = function() {
                    uploadedImage = img;
                    document.getElementById('uploadPrompt').style.display = 'none';
                    updateCanvas();
                    closeIconLibrary();
                    URL.revokeObjectURL(url);
                };
                
                img.src = url;
            } catch (error) {
                alert('Gagal memuat icon. Silakan coba lagi.');
            }
        }

        function searchIcons() {
            const searchTerm = document.getElementById('iconSearch').value.toLowerCase();
            
            if (searchTerm === '') {
                displayIcons(allIconsData);
            } else {
                const filtered = allIconsData.filter(icon => 
                    icon.name.toLowerCase().includes(searchTerm)
                );
                displayIcons(filtered);
            }
        }
